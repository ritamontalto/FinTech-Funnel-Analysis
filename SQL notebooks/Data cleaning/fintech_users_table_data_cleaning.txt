--- I check the phone number column data formatting

SELECT phone_number
FROM users;

SELECT phone_number
FROM users
WHERE phone_number IS NULL;


--- data is quite messy and not consistent here, it must be fixed. I start removing non numeric characters 

SELECT phone_number,
REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g') AS digits_only
FROM users;


--- then I standardize the format for all and define null values

SELECT phone_number,
CASE
    WHEN phone_number IS NULL THEN NULL
    WHEN phone_number = '' THEN NULL
    WHEN LEFT(REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g'), 2) = '44' 
    	THEN CONCAT('+', REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g'))
ELSE CONCAT('+44',RIGHT(REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g'), 10))
END AS clean_phone_number
FROM users;

--- I'm done with the data cleaning process. In order to keep the original dataset as is, I'm creating a new table with clean data that I can download and use for my analysis


CREATE TABLE users_clean AS 
SELECT user_id,
INITCAP(TRIM(first_name)) AS first_name,
INITCAP(TRIM(surname)) AS surname,
INITCAP(TRIM(full_name)) AS full_name,
CASE
    WHEN TRIM(city) = '' THEN NULL 
ELSE INITCAP(TRIM(city))
END AS city,
CASE
    WHEN TRIM(region) = '' THEN NULL 
ELSE INITCAP(TRIM(region))
END AS region,
CASE
    WHEN TRIM(country) IN ('UK', 'uk', 'U.K.', 'United kingdom') THEN 'United Kingdom'
	WHEN TRIM(country) = '' THEN NULL 
ELSE INITCAP(TRIM(country))
END AS country,
age,
CASE
	WHEN "age" BETWEEN 20 AND 35 THEN '20-35'
	WHEN "age" BETWEEN 36 AND 50 THEN '36-50'
	WHEN "age" BETWEEN 51 AND 65 THEN '51-65'
ELSE NULL
END AS age_range,
CASE 
	WHEN TRIM(gender) = '' THEN NULL 
ELSE INITCAP(TRIM(gender))
END AS gender,
income, 
CASE
	WHEN income = 0 THEN NULL 
	WHEN income < 15000 THEN 'Under 15k'
	WHEN income < 30000 THEN '15k-30k'
	WHEN income < 50000 THEN '30k-50k'
	WHEN income < 75000 THEN '50k-75k'
	WHEN income < 85000 THEN '75k-85k'
ELSE NULL
END AS income_range,
CASE 
	WHEN TRIM(employment_status) = '' THEN NULL 
ELSE TRIM(employment_status) 
END AS employment_status,
CASE 
	WHEN TRIM(marital_status) = '' THEN NULL 
ELSE TRIM(marital_status) 
END AS marital_status,
CASE 
	WHEN TRIM(email) = '' THEN NULL
	WHEN email NOT LIKE '%@%.%' THEN NULL
ELSE LOWER(TRIM(email))
END AS email,
CASE
    WHEN phone_number IS NULL THEN NULL
    WHEN phone_number = '' THEN NULL
    WHEN LEFT(REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g'), 2) = '44' 
    	THEN CONCAT('+', REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g'))
ELSE CONCAT('+44',RIGHT(REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g'), 10))
END AS phone_number,
CASE
	WHEN TRIM(acquisition_channel) = '' THEN NULL 
ELSE TRIM(acquisition_channel)
END AS acquisition_channel
FROM users;
