
--- EXPLORATORY BUSINESS ANALYSIS

--- The objective is to explore customer characteristics, application outcomes and portfolio performance to identify the factors that influence the loan application funnel before analysing conversion and drop-off throughout the customer journey.


--- SECTION 1: BUSINESS UNDERSTANDING

--- Q1: How many customers and applications do we have?

SELECT
COUNT(DISTINCT user_id) AS total_customers
FROM users_clean;


SELECT
COUNT(*) AS total_applications,
COUNT(DISTINCT user_id) AS unique_applicants
FROM loan_applications;



--- Q2: How are loan applications distributed across different outcomes?

SELECT outcome,
COUNT(*) AS applications
FROM loan_applications
GROUP BY outcome
ORDER BY applications DESC;

--- SECTION 3: CREDIT RISK ANALYSIS

--- Q14: How does estimated credit score vary across loan outcomes?

SELECT outcome,
ROUND(CAST(AVG(est_credit_score) AS NUMERIC), 1) AS avg_credit_score
FROM loan_applications
GROUP BY outcome
ORDER BY avg_credit_score DESC;



--- Q15: How does the requested loan amount differ across application outcomes?

SELECT outcome,
ROUND(CAST(AVG(requested_amount) AS NUMERIC), 2) AS avg_requested_amount
FROM loan_applications
GROUP BY outcome
ORDER BY avg_requested_amount DESC;




--- Q16: Which acquisition channels attract applicants with stronger credit profiles?

SELECT u.acquisition_channel,
ROUND(CAST(AVG(la.est_credit_score) AS NUMERIC), 1) AS avg_credit_score,
COUNT(*) AS applications
FROM loan_applications AS la
JOIN users_clean AS u
ON la.user_id = u.user_id
GROUP BY u.acquisition_channel
ORDER BY avg_credit_score DESC;



--- Q17: Which acquisition channels have the highest application approval rates?

SELECT u.acquisition_channel,
COUNT(*) AS applications,
SUM(CASE
    	WHEN la.outcome = 'Granted' THEN 1
        ELSE 0
        END) AS "granted",
ROUND(CAST(SUM(CASE
               		WHEN la.outcome = 'Granted' THEN 1
                	ELSE 0
            		END) AS NUMERIC) / COUNT(*) * 100, 2) AS approval_rate
FROM loan_applications AS la
JOIN users_clean AS u
ON la.user_id = u.user_id
GROUP BY u.acquisition_channel
ORDER BY approval_rate DESC;



--- SECTION 4: LOAN PORTFOLIO PERFORMANCE 

--- Q18: What percentage of granted loans were ultimately charged off? 

SELECT
COUNT(*) AS granted_loans,
SUM(charge_off_flag) AS charged_off_loans,
ROUND(CAST(AVG(charge_off_flag) AS NUMERIC) * 100, 2) AS charge_off_rate
FROM loan_outcomes
WHERE loan_status = 'Granted';



--- Q19: Which acquisition channels generate the highest proportion of charged-off loans? 

SELECT u.acquisition_channel,
COUNT(*) AS granted_loans,
SUM(lo.charge_off_flag) AS charge_offs,
ROUND(CAST(AVG(lo.charge_off_flag) AS NUMERIC) * 100, 2) AS charge_off_rate
FROM loan_outcomes AS lo
JOIN loan_applications AS la
ON lo.loan_id = la.loan_id
JOIN users_clean AS u
ON la.user_id = u.user_id
WHERE lo.loan_status = 'Granted'
GROUP BY u.acquisition_channel
ORDER BY charge_off_rate DESC;



--- Q20: How do charge-off rates vary across credit score bands? 

SELECT
CASE
	WHEN la.est_credit_score < 550 THEN '<550'
	WHEN la.est_credit_score < 650 THEN '550-649'
	WHEN la.est_credit_score < 750 THEN '650-749'
ELSE '750+'
END AS credit_score_band,
COUNT(*) AS granted_loans,
SUM(lo.charge_off_flag) AS charged_off_loans,
ROUND(CAST(AVG(lo.charge_off_flag) AS NUMERIC) * 100, 2) AS charge_off_rate
FROM loan_applications AS la
JOIN loan_outcomes AS lo
ON la.loan_id = lo.loan_id
WHERE lo.loan_status = 'Granted'
GROUP BY credit_score_band
ORDER BY credit_score_band;



--- Q21: How do average approved loan amounts vary across acquisition channels? 

SELECT u.acquisition_channel,
COUNT(*) AS granted_loans,
ROUND(CAST(AVG(lo.granted_amount) AS NUMERIC), 2) AS avg_granted_amount
FROM loan_outcomes AS lo
JOIN loan_applications AS la
ON lo.loan_id = la.loan_id
JOIN users_clean AS u
ON la.user_id = u.user_id
WHERE lo.loan_status = 'Granted'
GROUP BY u.acquisition_channel
ORDER BY avg_granted_amount DESC;



--- Q22: How does the final interest rate vary across different credit score bands?

SELECT
CASE
	WHEN est_credit_score < 550 THEN '<550'
	WHEN est_credit_score < 650 THEN '550-649'
	WHEN est_credit_score < 750 THEN '650-749'
ELSE '750+'
END AS score_band,
ROUND(CAST(AVG(lo.final_interest_rate) AS NUMERIC), 2) AS avg_interest_rate,
COUNT(*) AS loans
FROM loan_applications AS la
JOIN loan_outcomes AS lo
ON la.loan_id = lo.loan_id
WHERE lo.loan_status = 'Granted'
GROUP BY score_band
ORDER BY score_band;



--- Q23: How does the final interest rate vary across acquisition channels?
